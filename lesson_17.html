transform: scaleY(0.5);
transform: scaleZ(1.5);
transform: scale3d(0.8,1.5,1.5);

skew

transform: rotate(10, 10);

        0% {
          transform: scale(0);
        }
        100% {
          transform: scale(1);
        }
      }
